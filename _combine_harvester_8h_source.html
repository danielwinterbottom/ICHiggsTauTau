420 template <class Map>
421 void CombineHarvester::AddSyst(CombineHarvester& target,
422  std::string const& name, std::string const& type,
423  Map const& valmap) {
424  // Keep track of which Process entries get a Systematic assigned and which
425  // don't. If verbosity is on we'll print lists of these processes at the end.
426  std::vector<ch::Process *> not_added_procs;
427  std::vector<ch::Process *> added_procs;
428  // Also track which tuples in the map did not get used. Do this by getting the
429  // full map here and then removing elements as they are used to create a
430  // Systematic.
431  auto tuples = valmap.GetTupleSet();
432  if (verbosity_ >= 1) {
433  LOGLINE(log(), name + ":" + type);
434  }
435  for (unsigned i = 0; i < procs_.size(); ++i) {
436  if (!valmap.Contains(procs_[i].get())) {
437  not_added_procs.push_back(procs_[i].get());
438  continue;
439  }
440  tuples.erase(valmap.GetTuple(procs_[i].get()));
441  added_procs.push_back(procs_[i].get());
442  std::string subbed_name = name;
443  boost::replace_all(subbed_name, "$BIN", procs_[i]->bin());
444  boost::replace_all(subbed_name, "$PROCESS", procs_[i]->process());
445  boost::replace_all(subbed_name, "$MASS", procs_[i]->mass());
446  boost::replace_all(subbed_name, "$ERA", procs_[i]->era());
447  boost::replace_all(subbed_name, "$CHANNEL", procs_[i]->channel());
448  boost::replace_all(subbed_name, "$ANALYSIS", procs_[i]->analysis());
449  auto sys = std::make_shared<Systematic>();
450  ch::SetProperties(sys.get(), procs_[i].get());
451  sys->set_name(subbed_name);
452  sys->set_type(type);
453  if (type == "lnN") {
454  sys->set_asymm(valmap.IsAsymm());
455  sys->set_value_u(valmap.ValU(procs_[i].get()));
456  sys->set_value_d(valmap.ValD(procs_[i].get()));
457  } else if (type == "shape") {
458  sys->set_asymm(true);
459  sys->set_value_u(1.0);
460  sys->set_value_d(1.0);
461  sys->set_scale(valmap.ValU(procs_[i].get()));
462  }
463  CombineHarvester::CreateParameterIfEmpty(&target, sys->name());
464  target.systs_.push_back(sys);
465  }
466  if (tuples.size() && verbosity_ >= 1) {
467  log() << ">> Map keys that were not used to create a Systematic:\n";
468  for (auto s : tuples) {
469  log() << ch::Tuple2String(s) << "\n";
470  }
471  }
472  if (verbosity_ >= 2) {
473  Process::PrintHeader(log());
474  log() << ">> Process entries that did not get a Systematic:\n";
475  for (auto p : not_added_procs) {
476  log() << *p << "\n";
477  }
478  log() << ">> Process entries that did get a Systematic:\n";
479  for (auto p : added_procs) {
480  log() << *p << "\n";
481  }
482  }
483 }
484 }
