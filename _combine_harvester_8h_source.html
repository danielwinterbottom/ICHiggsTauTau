394 template <class Map>
395 void CombineHarvester::AddSyst(CombineHarvester& target,
396  std::string const& name, std::string const& type,
397  Map const& valmap) {
398  for (unsigned i = 0; i < procs_.size(); ++i) {
399  if (!valmap.Contains(procs_[i].get())) continue;
400  std::string subbed_name = name;
401  boost::replace_all(subbed_name, "$BIN", procs_[i]->bin());
402  boost::replace_all(subbed_name, "$PROCESS", procs_[i]->process());
403  boost::replace_all(subbed_name, "$MASS", procs_[i]->mass());
404  boost::replace_all(subbed_name, "$ERA", procs_[i]->era());
405  boost::replace_all(subbed_name, "$CHANNEL", procs_[i]->channel());
406  boost::replace_all(subbed_name, "$ANALYSIS", procs_[i]->analysis());
407  auto sys = std::make_shared<Systematic>();
408  ch::SetProperties(sys.get(), procs_[i].get());
409  sys->set_name(subbed_name);
410  sys->set_type(type);
411  if (type == "lnN") {
412  sys->set_asymm(valmap.IsAsymm());
413  sys->set_value_u(valmap.ValU(procs_[i].get()));
414  sys->set_value_d(valmap.ValD(procs_[i].get()));
415  } else if (type == "shape") {
416  sys->set_asymm(true);
417  sys->set_value_u(1.0);
418  sys->set_value_d(1.0);
419  sys->set_scale(valmap.ValU(procs_[i].get()));
420  }
421  CombineHarvester::CreateParameterIfEmpty(&target, sys->name());
422  target.systs_.push_back(sys);
423  }
424 }
