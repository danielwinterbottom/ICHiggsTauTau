00021 // CityHash, by Geoff Pike and Jyrki Alakuijala
00022 //
00023 // This file provides a few functions for hashing strings. On x86-64
00024 // hardware in 2011, CityHash64() is faster than other high-quality
00025 // hash functions, such as Murmur.  This is largely due to higher
00026 // instruction-level parallelism.  CityHash64() and CityHash128() also perform
00027 // well on hash-quality tests.
00028 //
00029 // CityHash128() is optimized for relatively long strings and returns
00030 // a 128-bit hash.  For strings more than about 2000 bytes it can be
00031 // faster than CityHash64().
00032 //
00033 // Functions in the CityHash family are not suitable for cryptography.
00034 //
00035 // WARNING: This code has not been tested on big-endian platforms!
00036 // It is known to work well on little-endian platforms that have a small penalty
00037 // for unaligned reads, such as current Intel and AMD moderate-to-high-end CPUs.
00038 //
00039 // By the way, for some hash functions, given strings a and b, the hash
00040 // of a+b is easily derived from the hashes of a and b.  This property
00041 // doesn't hold for any hash functions in this file.
00042

00080 // Hash 128 input bits down to 64 bits of output.
00081 // This is intended to be a reasonably good hash function.
00082 inline uint64 Hash128to64(const uint128& x) {
00083   // Murmur-inspired hashing.
00084   const uint64 kMul = 0x9ddfea08eb382d69ULL;
00085   uint64 a = (Uint128Low64(x) ^ Uint128High64(x)) * kMul;
00086   a ^= (a >> 47);
00087   uint64 b = (Uint128High64(x) ^ a) * kMul;
00088   b ^= (b >> 47);
00089   b *= kMul;
00090   return b;
00091 }
