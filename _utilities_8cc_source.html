129 std::vector<std::vector<unsigned>> GenerateCombinations(
130  std::vector<unsigned> vec) {
131  unsigned n = vec.size();
132  std::vector<unsigned> idx(n, 0);
133  std::vector<std::vector<unsigned>> result;
134  // if any one of the elements is zero there are no
135  // combinations to build
136  if (std::find(vec.begin(), vec.end(), 0) != vec.end()) {
137  return result;
138  }
139  result.push_back(idx);
140  bool exit_loop = false;
141  while (exit_loop == false) {
142  // Find the first index we can increment (if possible)
143  for (unsigned i = 0; i < n; ++i) {
144  if ((idx[i] + 1) == vec[i]) {
145  if (i != n - 1) {
146  idx[i] = 0;
147  } else {
148  // we're done
149  exit_loop = true;
150  break;
151  }
152  } else {
153  ++(idx[i]);
154  result.push_back(idx);
155  break;
156  }
157  }
158  }
159  return result;
160 }

179 std::vector<std::string> MassesFromRange(std::string const& input,
180  std::string const& fmt) {
181  std::set<double> mass_set;
182  std::vector<std::string> tokens;
183  boost::split(tokens, input, boost::is_any_of(","));
184  for (auto const& t : tokens) {
185  std::vector<std::string> sub_tokens;
186  boost::split(sub_tokens, t, boost::is_any_of("-:"));
187  if (sub_tokens.size() == 1) {
188  double mass_val = boost::lexical_cast<double>(sub_tokens[0]);
189  mass_set.insert(mass_val);
190  } else if (sub_tokens.size() == 3) {
191  double lo = boost::lexical_cast<double>(sub_tokens[0]);
192  double hi = boost::lexical_cast<double>(sub_tokens[1]);
193  double step = boost::lexical_cast<double>(sub_tokens[2]);
194  if (hi <= lo)
195  throw std::runtime_error(
196  "[MassesFromRange] High mass is smaller than low mass!");
197  double start = lo;
198  while (start < hi + 0.001) {
199  mass_set.insert(start);
200  start += step;
201  }
202  }
203  }
204  std::vector<std::string> result;
205  for (auto const& m : mass_set) {
206  result.push_back((boost::format(fmt) % m).str());
207  }
208  return result;
209 }
