70 void Observation::set_shape(std::unique_ptr<TH1> shape, bool set_rate) {
71  // We were given a nullptr - this is fine, and so we're done
72  if (!shape) {
73  // This will safely release any existing TH1 held by shape_
74  shape_ = nullptr;
75  return;
76  }
77  // Otherwise we validate this new hist. First is to check that all bins have
78  // +ve values, otherwise the interpretation as a pdf is tricky
79  for (int i = 1; i <= shape->GetNbinsX(); ++i) {
80  if (shape->GetBinContent(i) < 0.) {
81  throw std::runtime_error(FNERROR("TH1 has a bin with content < 0"));
82  }
83  }
84  // At this point we can safely move the shape in and take ownership
85  shape_ = std::move(shape);
86  // Ensure that root will not try and clean this up
87  shape_->SetDirectory(0);
88  if (set_rate) {
89  this->set_rate(shape_->Integral());
90  }
91  if (shape_->Integral() > 0.) shape_->Scale(1. / shape_->Integral());
92 }
